<script>

let paper = document.getElementById("paperRings");
let lover = document.getElementById("lover");
let engine = document.getElementById("engine");

function beginFlight(){

  // Switch screen
  document.getElementById("boarding").classList.remove("active");
  document.getElementById("cabin").classList.add("active");

  // Start sounds
  paper.volume = 0.4;
  paper.play();

  engine.volume = 0.2;
  engine.play();

  // Announcement after 8 seconds
  setTimeout(captainAnnouncement, 8000);

  // Emotional shift after 90 seconds
  setTimeout(shiftToLover, 90000);

  // Landing after 3.5 minutes
  setTimeout(showLanding, 210000);
}

function captainAnnouncement(){

  paper.volume = 0.2;

  const speech = new SpeechSynthesisUtterance(
  "Good evening passengers. This is your captain speaking. Today’s flight is dedicated to the most beautiful girl on board, Ashwini Parande. In flight refreshments have been placed in the dashboard for our passenger princess. We expect smooth skies on our journey to Lonavala.");

  speech.rate = 0.9;
  speech.pitch = 1;
  speechSynthesis.speak(speech);
}

function shiftToLover(){

  let fade = setInterval(function(){
    if(paper.volume > 0){
      paper.volume -= 0.02;
    }else{
      clearInterval(fade);
      paper.pause();
      lover.volume = 0;
      lover.play();
      fadeInLover();
    }
  },200);
}

function fadeInLover(){

  let fade = setInterval(function(){
    if(lover.volume < 0.5){
      lover.volume += 0.02;
    }else{
      clearInterval(fade);
    }
  },200);
}

function showLanding(){

  document.getElementById("cabin").classList.remove("active");
  document.getElementById("landing").classList.add("active");

  // Fade out music
  let fade = setInterval(function(){
    if(lover.volume > 0){
      lover.volume -= 0.02;
    }else{
      clearInterval(fade);
      lover.pause();
      engine.pause();
    }
  },200);
}

/* Interaction buttons */

  alert("Princess-approved KitKat minis located in dashboard.");
}

function seatbelt(){
  alert("Seatbelt secured. You’re stuck with me forever.");
}

function princess(){
  alert("Princess status: Confirmed. Priority boarding for life.");
}
